    #import "NSString+ValidNumber.h"
    @implementation NSString (ValidNumber)
    
    - (BOOL)isValidNumber
    {
        NSString *currentString = self;
    
        if (currentString.length == 0 || currentString.length > 100) {
            return false;
        }
        
        NSCharacterSet *digits = [NSCharacterSet decimalDigitCharacterSet];
    
        for (NSInteger i = 0; i < [currentString length]; i++)
        {
            unichar letter = [currentString characterAtIndex:i];
            if (![digits characterIsMember:letter])
            {
                return false;
            }
        }
        return true;
    }
    
    @end
